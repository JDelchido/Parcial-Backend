package com.finca.arriendo.controllers;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.CrossOrigin;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.finca.arriendo.dto.UsuarioDto;
import com.finca.arriendo.services.UsuarioService;

@RestController
@RequestMapping("/api/usuario")
public class UsuarioController {

    @Autowired
    private UsuarioService usuarioService;

    @CrossOrigin
    @GetMapping("/list")
    public List<UsuarioDto> get() throws Exception {
        return usuarioService.list();
    }

    @CrossOrigin
    @GetMapping("/get/{id}")
    public UsuarioDto get(@PathVariable Long id) throws Exception {
        return usuarioService.get(id);
    }

    @CrossOrigin
    @GetMapping("/getNom/{nombre}")
    public UsuarioDto get(@PathVariable String nombre) throws Exception {
        return usuarioService.get(nombre);
    }

    @CrossOrigin
    @PostMapping("/create")
    public UsuarioDto save(@RequestBody UsuarioDto usuarioDto) throws Exception {
        return usuarioService.saveNew(usuarioDto);
    }

    @CrossOrigin
    @PutMapping("/update")
    public UsuarioDto update(@RequestBody UsuarioDto usuarioDto) throws Exception {
        return usuarioService.save(usuarioDto);
    }

    @CrossOrigin
    @DeleteMapping("/delete")
    public void delete(@RequestBody UsuarioDto usuarioDto) throws Exception {
        usuarioService.delete(usuarioDto);
    }
}
