package com.finca.arriendo.controllers;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.CrossOrigin;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.finca.arriendo.dto.FincaDto;
import com.finca.arriendo.services.FincaService;

@RestController
@RequestMapping("/api/finca")
public class FincaController {
    
    @Autowired
    private FincaService fincaService;

    @CrossOrigin
    @GetMapping("/list")
    public List<FincaDto> get() throws Exception{
        return fincaService.list();
    }

    @CrossOrigin
    @GetMapping("/get/{id}")
    public FincaDto get(@PathVariable Long id) throws Exception{
        return fincaService.get(id);
    }

    @CrossOrigin
    @GetMapping("/getNom/{nombre}")
    public FincaDto obtenerPorNombre(@PathVariable String nombre) throws Exception{
        return fincaService.get(nombre);
    }

    @CrossOrigin
    @PostMapping("/create")
    public FincaDto save(@RequestBody FincaDto fincaDto) throws Exception{
        return fincaService.saveNew(fincaDto);
    }

    @CrossOrigin
    @PutMapping("/update")
    public FincaDto update(@RequestBody FincaDto fincaDto) throws Exception{
        return fincaService.save(fincaDto);
    }

    @CrossOrigin
    @DeleteMapping("/delete")
    public void delete(@RequestBody FincaDto fincaDto) throws Exception{
        fincaService.delete(fincaDto);
    }
}
