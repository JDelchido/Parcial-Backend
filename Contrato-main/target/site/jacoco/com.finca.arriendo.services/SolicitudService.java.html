package com.finca.arriendo.services;

import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

import org.modelmapper.ModelMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.finca.arriendo.dto.SolicitudDto;
import com.finca.arriendo.model.Estado;
import com.finca.arriendo.model.Solicitud;
import com.finca.arriendo.repository.SolicitudRepository;

@Service
public class SolicitudService {

    @Autowired
    SolicitudRepository solicitudRepository;
    @Autowired
    ModelMapper modelMapper;

    public SolicitudDto get(Long id){
        Optional<Solicitud> solicitud = solicitudRepository.findById(id);
        SolicitudDto solicitudDto = null;
        if(solicitud.isPresent()){
            solicitudDto = modelMapper.map(solicitud.get(), SolicitudDto.class);
        }
        return solicitudDto;
    }
    
    public List<SolicitudDto> list(){
        List<Solicitud> Lsolicitud = (List<Solicitud>) solicitudRepository.findAll();
        List<SolicitudDto> LsolicitudDtos = Lsolicitud.stream().map(solicitud -> modelMapper.map(solicitud, SolicitudDto.class))
                .collect(Collectors.toList());
        return LsolicitudDtos;
    }

    public SolicitudDto saveNew(SolicitudDto solicitudDto){
        Solicitud solicitud = modelMapper.map(solicitudDto, Solicitud.class);
        solicitud.setEstado(Estado.ACEPTADA);
        solicitud = solicitudRepository.save(solicitud);
        solicitudDto = modelMapper.map(solicitud, SolicitudDto.class);
        return solicitudDto;
    }

    public SolicitudDto save(SolicitudDto solicitudDto){
        Solicitud solicitud = modelMapper.map(solicitudDto, Solicitud.class);
        solicitud.setEstado(Estado.EN_TRAMITE);
        solicitud = solicitudRepository.save(solicitud);
        solicitudDto = modelMapper.map(solicitud, SolicitudDto.class);
        return solicitudDto;
    }

    public void delete(SolicitudDto solicitudDto){
        Solicitud solicitud = modelMapper.map(solicitudDto, Solicitud.class);
        solicitudRepository.delete(solicitud);
    }
}
