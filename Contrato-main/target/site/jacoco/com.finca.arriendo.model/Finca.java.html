package com.finca.arriendo.model;

import org.hibernate.annotations.SQLDelete;

import io.micrometer.common.lang.Nullable;
import jakarta.persistence.Entity;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

@Entity
@Getter
@Setter
@AllArgsConstructor
@NoArgsConstructor
@SQLDelete(sql = "UPDATE finca SET deleted = true WHERE id=?")
public class Finca {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne
    @JoinColumn(name = "arrendatario_id")
    private Usuario dueño;

    private String nombre;
    private String ubicacion;
    private String departamento;
    private String municipio;
    private float precio;
    private Boolean disponible;
    private int calificacion;

    @Nullable
    private String comentarios;

    private int capacidad;

    private boolean deleted = false;

    public Finca(Usuario dueño, String nombre, String ubicacion, String departamento, String municipio, float precio,
                 Boolean disponible, int calificacion, String comentarios, int capacidad, boolean deleted) {
        this.dueño = dueño;
        this.nombre = nombre;
        this.ubicacion = ubicacion;
        this.departamento = departamento;
        this.municipio = municipio;
        this.precio = precio;
        this.disponible = disponible;
        this.calificacion = calificacion;
        this.comentarios = comentarios;
        this.capacidad = capacidad;
        this.deleted = deleted;
    }

    public Finca(Usuario dueño, String nombre, String ubicacion, String departamento, String municipio, float precio,
                 Boolean disponible, int calificacion, int capacidad, boolean deleted) {
        this.dueño = dueño;
        this.nombre = nombre;
        this.ubicacion = ubicacion;
        this.departamento = departamento;
        this.municipio = municipio;
        this.precio = precio;
        this.disponible = disponible;
        this.calificacion = calificacion;
        this.capacidad = capacidad;
        this.deleted = deleted;
    }
}
