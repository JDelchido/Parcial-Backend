package com.finca.arriendo;

import org.modelmapper.ModelMapper;
import org.modelmapper.convention.MatchingStrategies;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import com.finca.arriendo.dto.FincaDto;
import com.finca.arriendo.dto.SolicitudDto;
import com.finca.arriendo.dto.UsuarioDto;
import com.finca.arriendo.model.Finca;
import com.finca.arriendo.model.Solicitud;
import com.finca.arriendo.model.Usuario;
@Configuration
public class ModelMapperConfiguration {

    @Bean
    public ModelMapper modelMapper() {
        ModelMapper modelMapper = new ModelMapper();
        modelMapper.getConfiguration().setMatchingStrategy(MatchingStrategies.STRICT);

        // Configuración para Usuario y UsuarioDto
        modelMapper.createTypeMap(Usuario.class, UsuarioDto.class).addMappings(mapper -> {
            mapper.map(Usuario::getId, UsuarioDto::setId);
            mapper.map(Usuario::getNombre, UsuarioDto::setNombre);
            mapper.map(Usuario::getApellido, UsuarioDto::setApellido);
            mapper.map(Usuario::getCorreo, UsuarioDto::setCorreo);
            mapper.map(Usuario::getTelefono, UsuarioDto::setTelefono);
            mapper.map(Usuario::getContrasena, UsuarioDto::setContrasena);
            mapper.map(Usuario::getCalificacion, UsuarioDto::setCalificacion);
            mapper.map(Usuario::getTipo, UsuarioDto::setTipo);
        });

        modelMapper.createTypeMap(UsuarioDto.class, Usuario.class).addMappings(mapper -> {
            mapper.map(UsuarioDto::getId, Usuario::setId);
            mapper.map(UsuarioDto::getNombre, Usuario::setNombre);
            mapper.map(UsuarioDto::getApellido, Usuario::setApellido);
            mapper.map(UsuarioDto::getCorreo, Usuario::setCorreo);
            mapper.map(UsuarioDto::getTelefono, Usuario::setTelefono);
            mapper.map(UsuarioDto::getContrasena, Usuario::setContrasena);
            mapper.map(UsuarioDto::getCalificacion, Usuario::setCalificacion);
            mapper.map(UsuarioDto::getTipo, Usuario::setTipo);
        });

        // Configuración para Solicitud y SolicitudDto
        modelMapper.createTypeMap(Solicitud.class, SolicitudDto.class).addMappings(mapper -> {
            mapper.map(Solicitud::getId, SolicitudDto::setId);
            mapper.map(Solicitud::getFechaEntrega, SolicitudDto::setFechaEntrega);
            mapper.map(Solicitud::getFechaSalida, SolicitudDto::setFechaSalida);
            mapper.map(Solicitud::getCalifFinca, SolicitudDto::setCalifFinca);
            mapper.map(Solicitud::getCalifArrendatario, SolicitudDto::setCalifArrendatario);
            mapper.map(Solicitud::getPrecio, SolicitudDto::setPrecio);
            mapper.map(Solicitud::getCantPersonas, SolicitudDto::setCantPersonas);
        });

        modelMapper.createTypeMap(SolicitudDto.class, Solicitud.class).addMappings(mapper -> {
            mapper.map(SolicitudDto::getId, Solicitud::setId);
            mapper.map(SolicitudDto::getFechaEntrega, Solicitud::setFechaEntrega);
            mapper.map(SolicitudDto::getFechaSalida, Solicitud::setFechaSalida);
            mapper.map(SolicitudDto::getCalifFinca, Solicitud::setCalifFinca);
            mapper.map(SolicitudDto::getCalifArrendatario, Solicitud::setCalifArrendatario);
            mapper.map(SolicitudDto::getPrecio, Solicitud::setPrecio);
            mapper.map(SolicitudDto::getCantPersonas, Solicitud::setCantPersonas);
        });

        // Configuración para Finca y FincaDto
        modelMapper.createTypeMap(Finca.class, FincaDto.class).addMappings(mapper -> {
            mapper.map(Finca::getId, FincaDto::setId);
            mapper.map(Finca::getNombre, FincaDto::setNombre);
            mapper.map(Finca::getUbicacion, FincaDto::setUbicacion);
            mapper.map(Finca::getDepartamento, FincaDto::setDepartamento);
            mapper.map(Finca::getMunicipio, FincaDto::setMunicipio);
            mapper.map(Finca::getPrecio, FincaDto::setPrecio);
            mapper.map(Finca::getDisponible, FincaDto::setDisponible);
            mapper.map(Finca::getCalificacion, FincaDto::setCalificacion);
            mapper.map(Finca::getComentarios, FincaDto::setComentarios);
            mapper.map(Finca::getCapacidad, FincaDto::setCapacidad);
        });

        modelMapper.createTypeMap(FincaDto.class, Finca.class).addMappings(mapper -> {
            mapper.map(FincaDto::getId, Finca::setId);
            mapper.map(FincaDto::getNombre, Finca::setNombre);
            mapper.map(FincaDto::getUbicacion, Finca::setUbicacion);
            mapper.map(FincaDto::getDepartamento, Finca::setDepartamento);
            mapper.map(FincaDto::getMunicipio, Finca::setMunicipio);
            mapper.map(FincaDto::getPrecio, Finca::setPrecio);
            mapper.map(FincaDto::getDisponible, Finca::setDisponible);
            mapper.map(FincaDto::getCalificacion, Finca::setCalificacion);
            mapper.map(FincaDto::getComentarios, Finca::setComentarios);
            mapper.map(FincaDto::getCapacidad, Finca::setCapacidad);
        });

        return modelMapper;
    }
}
